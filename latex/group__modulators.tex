\doxysubsubsubsection{\texorpdfstring{fluid\_mod\_mapping\_t}{fluid\_mod\_mapping\_t}}
{\footnotesize\ttfamily typedef double(\texorpdfstring{$\ast$}{*} fluid\+\_\+mod\+\_\+mapping\+\_\+t) (const \mbox{\hyperlink{group__Types_ga6c727efab500d6c0c350d4292e9aa5ef}{fluid\+\_\+mod\+\_\+t}} \texorpdfstring{$\ast$}{*}mod, int value, int range, int is\+\_\+src1, void \texorpdfstring{$\ast$}{*}data)}

This function transforms or maps a modulator source value into a normalized range of {\ttfamily \mbox{[}-\/1.\+0;+1.\+0\mbox{]}}.

See fluid\+\_\+mod\+\_\+set\+\_\+custom\+\_\+mapping().


\begin{DoxyParams}{Parameters}
{\em mod} & The modulator instance. The behavior is undefined if you modify {\ttfamily mod} through any of the {\ttfamily fluid\+\_\+mod\+\_\+set\texorpdfstring{$\ast$}{*}()} functions from within the callback. \\
\hline
{\em value} & The input value from the modulator source, which will be in range {\ttfamily \mbox{[}0;16383\mbox{]}} if the input source value is \doxylink{group__modulators_gga801993d69390b21a2d0311d44afb1932a72f5b8e63c997525574d15f83a5a2e24}{FLUID\+\_\+\+MOD\+\_\+\+PITCHWHEEL}, or {\ttfamily \mbox{[}0;127\mbox{]}} otherwise. \\
\hline
{\em range} & The value-\/range of the modulator source, i.\+e. {\ttfamily 16384}, if the input source value is \doxylink{group__modulators_gga801993d69390b21a2d0311d44afb1932a72f5b8e63c997525574d15f83a5a2e24}{FLUID\+\_\+\+MOD\+\_\+\+PITCHWHEEL}, otherwise {\ttfamily 128}. \\
\hline
{\em data} & Custom data pointer, as supplied via fluid\+\_\+mod\+\_\+set\+\_\+custom\+\_\+mapping(). \\
\hline
{\em is\+\_\+src1} & A boolean, which, if true, indicates that the mapping function is called for source1. Otherwise, it\textquotesingle{}s called for source2. Only useful if two sources have been specified with the \doxylink{group__modulators_gga4b2419150b2a167df7807d5020a6e996a7eddf66883e2c3d275ad9c1a1e715753}{FLUID\+\_\+\+MOD\+\_\+\+CUSTOM} flag set. \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}
A value mapped into range {\ttfamily \mbox{[}-\/1.\+0;+1.\+0\mbox{]}}. For return values that exceed the mentioned range, the behavior is unspecified (i.\+e. it may be honored, it may be clipped, ignored, the entire modulator may be disabled, etc.). 
\end{DoxyReturn}
